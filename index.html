window.onload = function() { if (isAdminURL) { const authed = sessionStorage.getItem('blood_admin_auth'); if (authed === '1') { showPage('admin'); renderStats(); renderTable(); } else showPage('login'); } else { showPage('register'); } }; function registerPerson() { const name = document.getElementById('r-name').value.trim(); const age = document.getElementById('r-age').value.trim(); const phone = document.getElementById('r-phone').value.trim(); const blood = document.getElementById('r-blood').value; const location = document.getElementById('r-location').value.trim(); const email = document.getElementById('r-email').value.trim(); if (!name || !age || !phone || !blood) { alert_('reg-alert', 'Please fill all required fields (name, age, phone, blood group).', 'error'); return; } const data = getData(); data.push({ id: Date.now(), name, age, phone, blood, location, email, date: new Date().toLocaleDateString('en-GB') }); saveData(data); alert_('reg-alert', 'Registration successful! Thank you for registering.', 'success'); clearForm(); } function clearForm() { ['r-name','r-age','r-phone','r-location','r-email'].forEach(id => document.getElementById(id).value = ''); document.getElementById('r-blood').value = ''; } function adminLogin() { const p = document.getElementById('admin-pass').value; if (p === PASS) { sessionStorage.setItem('blood_admin_auth', '1'); alert_('login-alert', '', 'error'); document.getElementById('admin-pass').value = ''; showPage('admin'); renderStats(); renderTable(); } else { alert_('login-alert', 'Incorrect password. Try again.', 'error'); } } function adminLogout() { sessionStorage.removeItem('blood_admin_auth'); showPage('login'); } function renderStats() { const data = getData(); const groups = ['A+','A-','B+','B-','AB+','AB-','O+','O-']; let html = `
${data.length}
Total donors
`; groups.forEach(g => { const cnt = data.filter(d => d.blood === g).length; if (cnt > 0) html += `
${cnt}
${g}
`; }); document.getElementById('stats-row').innerHTML = html; } function getFiltered() { const txt = document.getElementById('search-text').value.toLowerCase(); const bg = document.getElementById('search-blood').value; return getData().filter(d => { const match = !txt || [d.name, d.phone, d.location, d.email].some(v => (v||'').toLowerCase().includes(txt)); return match && (!bg || d.blood === bg); }); } function renderTable() { const rows = getFiltered(); const tbody = document.getElementById('donor-table'); const empty = document.getElementById('empty-msg'); if (!rows.length) { tbody.innerHTML = ''; empty.style.display = 'block'; return; } empty.style.display = 'none'; tbody.innerHTML = rows.map((d, i) => ` ${i+1} ${d.name} ${d.age} ${d.blood} ${d.phone} ${d.location||'—'} ${d.email||'—'} ${d.date}
`).join(''); } function deleteRecord(id) { if (!confirm('Delete this record?')) return; saveData(getData().filter(d => d.id !== id)); renderStats(); renderTable(); } function shareRecord(id) { const d = getData().find(r => r.id === id); if (!d) return; const text = `Blood Donor Info\nName: ${d.name}\nAge: ${d.age}\nBlood Group: ${d.blood}\nPhone: ${d.phone}\nCity: ${d.location||'N/A'}`; if (navigator.share) navigator.share({ title: 'Blood Donor', text }); else if (navigator.clipboard) navigator.clipboard.writeText(text).then(() => alert('Copied to clipboard!')); else alert(text); } function exportExcel() { const rows = getFiltered(); if (!rows.length) { alert('No records to export.'); return; } const ws = XLSX.utils.json_to_sheet(rows.map((d,i) => ({ '#': i+1, 'Name': d.name, 'Age': d.age, 'Blood Group': d.blood, 'Phone': d.phone, 'City': d.location||'', 'Email': d.email||'', 'Registered': d.date }))); const wb = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(wb, ws, 'Donors'); XLSX.writeFile(wb, 'blood_registry.xlsx'); } function exportPDF() { const rows = getFiltered(); if (!rows.length) { alert('No records to export.'); return; } const { jsPDF } = window.jspdf; const doc = new jsPDF({ orientation: 'landscape' }); doc.setFontSize(16); doc.text('Blood Group Registry', 14, 16); doc.setFontSize(10); doc.setTextColor(120); doc.text(`Generated: ${new Date().toLocaleDateString('en-GB')} | Total: ${rows.length}`, 14, 23); doc.autoTable({ startY: 28, head: [['#','Name','Age','Blood','Phone','City','Email','Registered']], body: rows.map((d,i) => [i+1, d.name, d.age, d.blood, d.phone, d.location||'', d.email||'', d.date]), styles: { fontSize: 9, cellPadding: 3 }, headStyles: { fillColor: [226, 75, 74], textColor: 255, fontStyle: 'bold' }, alternateRowStyles: { fillColor: [252, 235, 235] } }); doc.save('blood_registry.pdf'); }
